## Install

```bash
npm install --save multi-filters
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'multi-filters'
import 'multi-filters/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
